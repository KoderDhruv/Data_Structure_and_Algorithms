
  class Solution{
    public static long divide(long a, long b){
        long sign = ((a<0 && b>=0) || (a>=0 && b<0))?-1L:1L;
        a = Math.abs(a);
        b = Math.abs(b);
        long ans = 0L;
        for(int i = 31; i>=0; i--){
            if((b<<i)<=a){
                a -= b<<i;
                ans |= 1L<<i;
            }
        }
        return sign<0?-ans:ans;
    }
}
